        return False
    
    try:

[docs]def extract_numbers(string):
    pattern = r'(\d+)(?:-(\d+))?$'
    match = re.match(pattern, string)
    if match:

        if not match.group(2):
            return 0, int(match.group(1))

        start = int(match.group(1))
        end = int(match.group(2))

        return start, end
    else:
        return None
